vue router 路由模块化

# 文件结构

src  
+-- router  
|   +-- modules 路由模块化管理文件夹  
|   |   +-- home.js 首页路由配置文件  
|   +-- index.js 路由文件

# router/index.js 核心文件

import Vue from 'vue'
import Router from 'vue-router'

import homeRouter from '@/router/modules/home'

Vue.use(Router)

let routes = []

// 主要页面配置 index
let rootRouter = {
  path: '/',
  component: () => import('pages/index/index'),
  children: [],
  redirect: '/home'
}
// 未找到指定路由调回主页面,可以考虑配置 404 页面
let redirectRouter = {
  path: '*',
  redirect: '/home'
}

homeRouter(rootRouter.children)
// 如有多个模块，依次在这里配置

const router = new Router({
  mode: 'history',
  routes: routes.concat([rootRouter, redirectRouter])
})
export default router

# router/modules/home.js

// 配置 主页 home
export default function (router) {
  router.push({
    path: '/home',
    name: 'home',
    component: () => import('pages/home/home'),
  })

  // 页面配置
  let routerObj = {
    homeQRCode: '首页 扫码注册',
    homePayInfo: '首页 支付查询',
    homeEarnings: '首页 收益提现',
    homeEarningsInfo: '首页 收益提现 设置提现资料',
    homeEarningsDetail: '首页 收益提现 提现明细',
    homeAddress: '首页 地址管理',
  }

  for (const key in routerObj) {
    if (routerObj.hasOwnProperty(key)) {
      router.push({
        path: '/' + key,
        name: key,
        component: () => import('pages/home/components/' + key + '/' + key)
      })
    }
  }

  // 当需要自定义路由按下列方式配置
  // router.push({
  //   path: '/test',
  //   name: 'test',
  //   component: () => import('pages/home/components/test/test')
  // })

}
