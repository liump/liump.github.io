在 vue 中使用百度地图

# 注册 key 百度地图开发者平台

# 原生百度地图

  1. vue2.X 配置webpack
let webpackConfig = {
  context: path.resolve(__dirname, '../'),
  entry: {
    app: './src/main.js'
  },
	// 配置非依赖包
  externals: {
    'BMap': 'BMap'
  },
}
  1. 创建组件
  • 创建地图功能文件夹
  • 创建 bmap.vue
<template>
  <div>
    <div id="bmap"></div>
  </div>
</template>

<script>
export default {
  name: 'bmap',
  data() {
    return {
      bmap: null,
      point: ''
    }
  },
  methods: {
    init() {
      this.bmap = new BMap.Map('bmap')
      this.point = new BMap.Point(116.404, 39.915)
      this.bmap.centerAndZoom(this.point, 14)
    }
  },
  activated(){
    this.init()
  }
};
</script>

<style lang="less" scoped>
#bmap {
  width: 100%;
  height: 100%;
  position: absolute;
}
</style>

# 使用Vue Baidu Map

# Vue Baidu Map中文文档

# CDN 引入

  • index.html中引入 <script src="https://unpkg.com/vue-baidu-map"></script>
  • main.js 中添加
Vue.use(VueBaiduMap.default, {
	// ak 的值为注册的地图 key
  ak: 'YOUR_KEY'
})
  • bmap.vue
<template>
  <div>
    <baidu-map class="bmap" :center="center" :zoom="zoom" @ready="handler"></baidu-map>
  </div>
</template>

<script>
export default {
  name: 'bmap',
  data() {
    return {
      center: { lng: 0, lat: 0 },
      zoom: 3
    };
  },
  methods: {
    handler({ BMap, map }) {
      console.log(BMap, map);
      this.center.lng = 116.404;
      this.center.lat = 39.915;
      this.zoom = 15;
    }
  }
};
</script>

<style lang="less" scoped>
.bmap {
  width: 100%;
  height: 100%;
  position: absolute;
}
</style>
